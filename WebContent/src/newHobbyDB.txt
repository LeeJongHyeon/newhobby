CREATE TABLE MEMBER 
(
MEMBER_ID VARCHAR(20) PRIMARY KEY, 

PWD VARCHAR(20),
 
NAME VARCHAR(20),

GENDER VARCHAR(20),

ZIPCODE VARCHAR(10),

ADDR1 VARCHAR(20), 

ADDR2 VARCHAR(30), 
   
REGDATE DATETIME DEFAULT NOW()

);




CREATE TABLE HOBBY
(
-- MySQL에서 AUTO_INCREMENT 가 ORACLE의 Sequence 역할.
	
HOBBY_IDX INT(6) PRIMARY KEY AUTO_INCREMENT,
	
MEMBER_ID VARCHAR(20),
	
TITLE VARCHAR(20), 
	
CONTENT VARCHAR(30), 
	
GENDER VARCHAR(20), 
	
HOBBY VARCHAR(40), 
	
IMG_FILE_NAME VARCHAR(40),
    
REGDATE DATETIME DEFAULT NOW()

);



-- MEMBER_ID에 대한 Foreign Key(참조키) 제약조건 추가

ALTER TABLE HOBBY
	
ADD CONSTRAINT FK_HOBBY_MEMBER_ID
	
Foreign KEY ( MEMBER_ID ) 
	
REFERENCES MEMBER ( MEMBER_ID )
	
ON DELETE CASCADE;





CREATE TABLE HOBBY_REPLY
(
RHOBBY_IDX INT(4) PRIMARY KEY AUTO_INCREMENT, 
	
RWRITER VARCHAR(20), 
	
RSUBJECT VARCHAR(20), 
	
RMEMO VARCHAR(30), 
	
HOBBY_IDX INT(4),
    
REGDATE DATETIME DEFAULT NOW()

);



-- HOBBY_IDX에 대한 Foreign Key(참조키) 제약조건

ALTER TABLE HOBBY_REPLY
	
ADD CONSTRAINT FK_HOBBY_REPLY_HOBBY_IDX
	
Foreign KEY ( HOBBY_IDX ) 
	
REFERENCES HOBBY ( HOBBY_IDX )
	
ON DELETE CASCADE;
